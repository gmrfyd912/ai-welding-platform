    const zoomControls = document.getElementById('zoom-controls');
    let isDragging = false; let dragOffsetX = 0, dragOffsetY = 0;

    function startDrag(e) {
        isDragging = true;
        let clientX = e.type.includes('mouse') ? e.clientX : e.touches[0].clientX;
        let clientY = e.type.includes('mouse') ? e.clientY : e.touches[0].clientY;
        const rect = zoomControls.getBoundingClientRect();
        dragOffsetX = clientX - rect.left; dragOffsetY = clientY - rect.top;
        zoomControls.style.right = 'auto'; zoomControls.style.bottom = 'auto'; 
    }
    function doDrag(e) {
        if (!isDragging) return;
        e.preventDefault(); 
        let clientX = e.type.includes('mouse') ? e.clientX : e.touches[0].clientX;
        let clientY = e.type.includes('mouse') ? e.clientY : e.touches[0].clientY;
        let newX = clientX - dragOffsetX; let newY = clientY - dragOffsetY;
        newX = Math.max(0, Math.min(window.innerWidth - zoomControls.offsetWidth, newX));
        newY = Math.max(0, Math.min(window.innerHeight - zoomControls.offsetHeight, newY));
        zoomControls.style.left = newX + 'px'; zoomControls.style.top = newY + 'px';
    }
    function endDrag() { isDragging = false; }
    zoomControls.addEventListener('mousedown', startDrag); window.addEventListener('mousemove', doDrag); window.addEventListener('mouseup', endDrag);
    zoomControls.addEventListener('touchstart', startDrag, {passive: false}); window.addEventListener('touchmove', doDrag, {passive: false}); window.addEventListener('touchend', endDrag);

        gameState = {

        towers = []; enemies = []; projectiles = []; particles = []; baseCracks = [];
        quizBag = [...allQuizzes].sort(() => Math.random() - 0.5);

        for(let i=0; i<25; i++) {
            let pts = [{x: 5 + Math.random()*50, y: 30 + Math.random()*620}];
            let segments = 3 + Math.floor(Math.random()*4);
            let angle = (Math.random() - 0.5) * Math.PI;
            for(let j=0; j<segments; j++) {
                angle += (Math.random()-0.5) * 1.5; 
                let len = 8 + Math.random()*20;
                let nx = pts[pts.length-1].x + Math.cos(angle)*len;
                let ny = pts[pts.length-1].y + Math.sin(angle)*len;
                nx = Math.max(2, Math.min(75, nx)); 
                pts.push({x: nx, y: ny});
            }
            baseCracks.push({ pts: pts, thresh: 0.1 + Math.random() * 0.8 });
        }
        updateUI();
        document.getElementById('btn-speed').innerText = '▶ 속도 x1';
        document.getElementById('btn-playpause').innerText = '일시정지';
        document.getElementById('btn-playpause').className = 'ctrl-btn btn-play';
    }

    function drawEnemy(ctx, e) {
        ctx.translate(e.x, e.y); ctx.fillStyle = e.color;
        if (e.type === 'Spatter') { ctx.beginPath(); ctx.arc(0, 0, e.radius, 0, Math.PI*2); ctx.fill(); } 
        else if (e.type === 'Porosity') {
            ctx.fillRect(-e.radius, -e.radius, e.radius*2, e.radius*2); ctx.fillStyle = '#111';
            ctx.beginPath(); ctx.arc(-5, -5, 2.5, 0, Math.PI*2); ctx.fill();
            ctx.beginPath(); ctx.arc(5, 5, 2.5, 0, Math.PI*2); ctx.fill();
            ctx.beginPath(); ctx.arc(-4, 4, 1.5, 0, Math.PI*2); ctx.fill();
        } else if (e.type === 'Undercut') {
            ctx.fillRect(-e.radius, -e.radius, e.radius*2, e.radius*2);
            ctx.fillStyle = '#111'; ctx.fillRect(-e.radius, -2, e.radius*2, 4);
        } else if (e.type === 'Overlap') {
            ctx.beginPath(); ctx.arc(-e.radius/2.5, 0, e.radius*0.7, 0, Math.PI*2); ctx.fill();
            ctx.beginPath(); ctx.arc(e.radius/2.5, 0, e.radius*0.7, 0, Math.PI*2); ctx.fill();
        } else if (e.type === 'Fusion') {
            // ★ 신규 결함: 용착불량 (찌그러진 다각형)
            ctx.beginPath();
            let r = e.radius;
            ctx.moveTo(-r, -r + 2);
            ctx.lineTo(0, -r - 4);
            ctx.lineTo(r, -r + 3);
            ctx.lineTo(r + 4, 0);
            ctx.lineTo(r - 2, r);
            ctx.lineTo(-2, r + 5);
            ctx.lineTo(-r, r - 1);
            ctx.lineTo(-r - 3, 0);
            ctx.closePath();
            ctx.fill();
            ctx.strokeStyle = '#111'; ctx.lineWidth = 1.5; ctx.stroke();
        } else if (e.type === 'Crack') {
            ctx.fillRect(-e.radius, -e.radius, e.radius*2, e.radius*2);
            ctx.strokeStyle = '#111'; ctx.lineWidth = 4;
            ctx.beginPath(); ctx.moveTo(-e.radius, -e.radius/2); ctx.lineTo(-e.radius/2, 2); ctx.lineTo(e.radius/2, -8); ctx.lineTo(e.radius, e.radius/2); ctx.stroke();
        }
        ctx.fillStyle = '#ecf0f1'; ctx.font = '12px Malgun Gothic bold'; ctx.textAlign = 'center';
        ctx.fillText(e.label, 0, -e.radius - 8); ctx.textAlign = 'left';
        ctx.translate(-e.x, -e.y);
    }

    canvas.addEventListener('click', (e) => {
        if (gameState.status !== 'playing') return;
        const rect = canvas.getBoundingClientRect();
        
        const mx = (e.clientX - rect.left) / currentScale; 
        const my = (e.clientY - rect.top) / currentScale;

        if(mx < 90) { openBaseMenu(mx, my); return; }
        
        let clickedTower = towers.find(t => Math.hypot(t.x - mx, t.y - my) < 35);
        if (clickedTower) { openUpgradeMenu(clickedTower, mx, my); } 
        else {
            if (!towers.some(t => Math.hypot(t.x - mx, t.y - my) < 30)) { openBuildMenu(mx, my); } 
            else { closeMenus(); }
        }
    });

    function positionMenu(menu, mx, my) {
        menu.style.display = 'flex';
        let px = mx + 15, py = my + 15;
        if (px + 250 > 1040) px = mx - 265;
        if (py + 250 > 680) py = my - 265;
        menu.style.left = `${px}px`; menu.style.top = `${py}px`;
    }

    function openBuildMenu(mx, my) { closeMenus(); gameState.buildTargetPos = {x: mx, y: my}; positionMenu(document.getElementById('build-menu'), mx, my); }
    function openUpgradeMenu(tower, mx, my) { closeMenus(); gameState.selectedTower = tower; refreshUpgradeMenu(); positionMenu(document.getElementById('upgrade-menu'), mx, my); }
    function openBaseMenu(mx, my) { closeMenus(); refreshBaseMenu(); positionMenu(document.getElementById('base-menu'), mx, my); }
    function closeMenus() { document.getElementById('build-menu').style.display = 'none'; document.getElementById('upgrade-menu').style.display = 'none'; document.getElementById('base-menu').style.display = 'none'; gameState.selectedTower = null; gameState.buildTargetPos = null; }

    window.quitGame = function() { if(confirm('게임을 종료하시겠습니까?')) triggerGameOver(true); }

    window.buyBaseUpgrade = function(type) {
        let cost = getBaseUpgCost(type);
        if (gameState.resources >= cost) {
            gameState.resources -= cost;
            if(type === 'alloy') { gameState.baseHp = Math.min(gameState.baseMaxHp, gameState.baseHp + 50); } 
            else if (type === 'bead') { gameState.beadShield += 100; } 
            else if (type === 'notch') { gameState.notchLv++; } 
            else if (type === 'heat') { gameState.heatLv++; }
            updateUI(); refreshBaseMenu();
        }
    }
    function getBaseUpgCost(type) { if(type==='alloy') return 80; if(type==='bead') return 100; if(type==='notch') return 150 + (gameState.notchLv * 50); if(type==='heat') return 120 + (gameState.heatLv * 40); }
    function refreshBaseMenu() {
        document.getElementById('btn-base-alloy').disabled = (gameState.resources < 80 || gameState.baseHp >= gameState.baseMaxHp);
        document.getElementById('btn-base-bead').disabled = (gameState.resources < 100);
        document.getElementById('cost-base-notch').innerText = getBaseUpgCost('notch'); document.getElementById('btn-base-notch').disabled = (gameState.resources < getBaseUpgCost('notch'));
        document.getElementById('cost-base-heat').innerText = getBaseUpgCost('heat'); document.getElementById('btn-base-heat').disabled = (gameState.resources < getBaseUpgCost('heat'));
    }

    // ★ 로봇 체력 속성 부여 (hp: 100)
    window.buildTower = function(type) {
        if (gameState.resources >= TOWER_SPECS[type].cost) {
            gameState.resources -= TOWER_SPECS[type].cost;
            towers.push({ 
                x: gameState.buildTargetPos.x, y: gameState.buildTargetPos.y, 
                type: type, hp: 100, maxHp: 100, cooldown: 0, angle: 0, 
                upg: { current: 0, voltage: 0, gas: 0, filler: 0 }, 
                torch_end: {x: gameState.buildTargetPos.x, y: gameState.buildTargetPos.y} 
            });
            updateUI(); closeMenus();
        }
    };

        let t = gameState.selectedTower; if(!t) return;
        let cost = getUpgCost(upgType, t.upg[upgType]);
        if (gameState.resources >= cost) { gameState.resources -= cost; t.upg[upgType]++; updateUI(); refreshUpgradeMenu(); }
    };
    
    function getTowerTotalCost(t) {
        let total = TOWER_SPECS[t.type].cost;
        let types = ['current', 'voltage', 'gas', 'filler'];
        types.forEach(type => { for(let i=0; i<t.upg[type]; i++) total += getUpgCost(type, i); });
        return total;
    }
    
    window.sellTower = function() {
        let t = gameState.selectedTower; if(!t) return;
        let refund = Math.floor(getTowerTotalCost(t) / 2);
        gameState.resources += refund;
        towers = towers.filter(tower => tower !== t);
        createFloatingText(t.x, t.y, `철거 +${refund}⚡`, "#f1c40f", 16);
        updateUI(); closeMenus();
    }

    function refreshUpgradeMenu() {
        let t = gameState.selectedTower; if(!t) return;
        document.getElementById('upg-title').innerText = `${t.type} 로봇 (총 Lv.${t.upg.current + t.upg.voltage + t.upg.gas + t.upg.filler})`;
        ['current', 'voltage', 'gas', 'filler'].forEach(type => { let c = getUpgCost(type, t.upg[type]); document.getElementById(`cost-${type}`).innerText = c; document.getElementById(`btn-upg-${type}`).disabled = (gameState.resources < c); });
        document.getElementById('refund-amt').innerText = `+${Math.floor(getTowerTotalCost(t)/2)}⚡`;
    }

    function updateUI() {
        document.getElementById('resources').innerText = gameState.resources;
        document.getElementById('base-hp').innerText = Math.floor(gameState.baseHp);
        document.getElementById('bead-hp-ui').innerText = gameState.beadShield > 0 ? ` (+비드보호: ${gameState.beadShield})` : '';
        document.getElementById('wave-num').innerText = gameState.wave;
        document.getElementById('wave-timer').innerText = Math.max(0, Math.ceil(gameState.waveTimerFrames / 60));
        document.getElementById('quiz-total').innerText = gameState.quizTotal;
        document.getElementById('quiz-correct').innerText = gameState.quizCorrect;
        document.getElementById('regen-fill').style.width = gameState.regenProgress + '%';
    }

    function getNextQuiz() {
        if(quizBag.length === 0) { quizBag = [...allQuizzes].sort(() => Math.random() - 0.5); }
        return quizBag.pop();
    }

    function triggerQuiz() {
        gameState.status = 'paused'; 
        currentQuiz = getNextQuiz();
        document.getElementById('quiz-cat-ui').innerText = `[ ${currentQuiz.cat} ]`;
        document.getElementById('quiz-text').innerText = "Q. " + currentQuiz.q;
        document.getElementById('quiz-exp').style.display = 'none';
        document.getElementById('quiz-buttons').style.display = 'block';
        document.getElementById('btn-next-wave').style.display = 'none';
        document.getElementById('quiz-overlay').style.display = 'flex';
        closeMenus();
    }
    
    // ★ 무조건 해설(exp)을 노출하도록 수정 완료
    window.answerQuiz = function(userAnswer) {
        document.getElementById('quiz-buttons').style.display = 'none';
        gameState.quizTotal++;
        
        let expText = currentQuiz.exp ? `\n\n[해설]\n${currentQuiz.exp}` : "";

        if (userAnswer === currentQuiz.a) {
            gameState.resources += 100;
            gameState.quizCorrect++;
            document.getElementById('quiz-exp').innerText = `✅ 정답입니다! (자원 +100)${expText}`;
            document.getElementById('quiz-exp').style.color = '#2ecc71';
        } else {
            gameState.penaltyActive = true;
            document.getElementById('quiz-exp').innerText = `❌ 오답입니다! (이번 웨이브 적 능력치 대폭 강화)${expText}`;
            document.getElementById('quiz-exp').style.color = '#e74c3c';
        }
        document.getElementById('quiz-exp').style.display = 'block'; 
        document.getElementById('btn-next-wave').style.display = 'block'; 
        updateUI();
    };
    
    window.resumeGame = function() { 
        document.getElementById('quiz-overlay').style.display = 'none'; 
        gameState.status = 'playing'; 
        requestAnimationFrame(gameLoop); 
    };

    function startNextWave() {
        gameState.wave++;
        gameState.waveTimerFrames = 30 * 60; 
        gameState.bossSpawned = false;
        gameState.penaltyActive = false;
        gameState.midQuizTriggered = false; // ★ 중간 퀴즈 플래그
        
        // ★ 용착불량 스폰 횟수 세팅
        gameState.fusionToSpawn = gameState.wave < 10 ? 10 : gameState.wave;
        
        triggerQuiz();
    }

    function spawnNormalEnemy(isFusion = false) {
        let type;
        if (isFusion) {
            type = 'Fusion';
        } else {
            let typesAvailable = ['Spatter'];
            if(gameState.wave >= 2) typesAvailable.push('Porosity');
            if(gameState.wave >= 4) typesAvailable.push('Undercut');
            if(gameState.wave >= 6) typesAvailable.push('Overlap');
            type = typesAvailable[Math.floor(Math.random() * typesAvailable.length)];
        }
        let spec = ENEMY_SPECS[type];
        
        // 난이도 상승 (체력 & 속도)
        let diffMult = 1.0 + (gameState.wave * 0.15);
        let speedMult = 1.0 + (gameState.wave * 0.02);
        let totalHpMult = diffMult * (gameState.penaltyActive ? 1.5 : 1.0); 
        
        enemies.push({ 
            x: canvas.width + 20, y: 80 + Math.random() * (canvas.height - 160), 
            type: type, label: spec.label, hp: spec.hp * totalHpMult, maxHp: spec.hp * totalHpMult, 
            speed: spec.speed * speedMult * (Math.random()*0.4 + 0.8),
            radius: spec.radius, color: spec.color, reward: spec.reward, dmg: spec.dmg, slowTimer: 0 
        });
    }

    function spawnBossSwarm() {
        let spec = ENEMY_SPECS['Crack'];
        let diffMult = 1.0 + (gameState.wave * 0.15);
        let totalHpMult = diffMult * (gameState.penaltyActive ? 1.5 : 1.0); 
        
        let bossCount = 1 + Math.floor(gameState.wave / 10);
        
        let swarmTypes = ['Spatter'];
        if (gameState.wave >= 5) swarmTypes.push('Porosity');
        if (gameState.wave >= 15) swarmTypes.push('Undercut');
        if (gameState.wave >= 25) { swarmTypes.push('Overlap'); swarmTypes.shift(); } 

        for(let b=0; b<bossCount; b++) {
            let bossX = canvas.width + 80 + (b * 150); 
            let bossY = canvas.height / 2 + ((b - Math.floor(bossCount/2)) * 80);

            let bossObj = { 
                id: 'boss_' + gameState.frameCount + '_' + b,
                x: bossX, y: bossY, 
                type: 'Crack', label: spec.label, hp: spec.hp * totalHpMult, maxHp: spec.hp * totalHpMult, 
                speed: spec.speed * (1.0 + gameState.wave * 0.01), radius: spec.radius, color: spec.color, reward: spec.reward, dmg: spec.dmg, slowTimer: 0 
            };
            enemies.push(bossObj);
            
            let swarmCount = 8 + Math.floor(gameState.wave * 0.6);
            if(swarmCount > 24) swarmCount = 24; 

            for(let i=0; i<swarmCount; i++) {
                let sType = swarmTypes[Math.floor(Math.random() * swarmTypes.length)];
                let sSpec = ENEMY_SPECS[sType];
                
                let angle = (i / swarmCount) * Math.PI * 2;
                let dist = bossObj.radius + sSpec.radius + 15 + Math.random() * 10;
                let offsetX = Math.cos(angle) * dist;
                let offsetY = Math.sin(angle) * dist;

                enemies.push({
                    x: bossX + offsetX, y: bossY + offsetY, 
                    type: sType, label: sSpec.label, hp: sSpec.hp * totalHpMult, maxHp: sSpec.hp * totalHpMult, 
                    speed: bossObj.speed, 
                    radius: sSpec.radius, color: sSpec.color, reward: sSpec.reward, dmg: sSpec.dmg, slowTimer: 0,
                    bossRef: bossObj, offsetX: offsetX, offsetY: offsetY 
                });
            }
        }
    }

    function drawAndCalculateRobot(ctx, t, spec, actualRange) {
        let totalLv = t.upg.current + t.upg.voltage + t.upg.gas + t.upg.filler;
        let scale = 0.33 + (totalLv * 0.015); 
        const L1 = 45, L2 = 65, L3 = 55, J4L = 15;
        const J1 = 0, J2 = -Math.PI/6, J3 = Math.PI/3, J4 = -Math.PI/6;

        let cx = 0, cy = 0, ang = 0;
        ang+=J1; cx+=L1*Math.cos(ang); cy+=L1*Math.sin(ang);
        ang+=J2; cx+=L2*Math.cos(ang); cy+=L2*Math.sin(ang);
        ang+=J3; cx+=L3*Math.cos(ang); cy+=L3*Math.sin(ang);
        ang+=J4; cx+=J4L*Math.cos(ang); cy+=J4L*Math.sin(ang);

        let J5_wx = t.x + cx * scale; let J5_wy = t.y + cy * scale;
        let torch_l = 10 + (t.type === 'FCAW' ? 35 : 15);
        t.torch_end = { x: J5_wx + Math.cos(t.angle) * torch_l * scale, y: J5_wy + Math.sin(t.angle) * torch_l * scale };

        ctx.save(); ctx.translate(t.x, t.y); ctx.scale(scale, scale);
        ctx.fillStyle = '#34495e'; ctx.fillRect(-12, -12, 24, 24); ctx.fillStyle = '#95a5a6'; ctx.beginPath(); ctx.arc(0,0,10,0,Math.PI*2); ctx.fill();

        let rang = 0;
        const drawLink = (len) => { ctx.fillStyle = '#34495e'; ctx.rotate(rang); ctx.fillRect(0, -3.5, len, 7); ctx.translate(len, 0); ctx.fillStyle = spec.color; ctx.beginPath(); ctx.arc(0,0,8,0,Math.PI*2); ctx.fill(); };
        rang = J1; drawLink(L1); rang = J2; drawLink(L2); rang = J3; drawLink(L3);
        ctx.fillStyle = '#34495e'; rang = J4; ctx.rotate(rang); ctx.fillRect(0, -2.5, J4L, 5); ctx.translate(J4L, 0); ctx.fillStyle = spec.color; ctx.beginPath(); ctx.arc(0,0,5,0,Math.PI*2); ctx.fill();

        ctx.rotate(t.angle - (J1+J2+J3+J4)); 
        if (t.type === 'FCAW') { ctx.fillStyle = '#f39c12'; ctx.fillRect(5, -5, 15, 10); ctx.beginPath(); ctx.arc(10, -5, 8, 0, Math.PI*2); ctx.fillStyle = '#ecf0f1'; ctx.fill(); ctx.fillStyle = '#7f8c8d'; ctx.fillRect(20, -2.5, 10, 5); ctx.fillRect(30, -2.5, 5, 15); } 
        else { ctx.fillStyle = '#f1c40f'; ctx.fillRect(5, -4, 15, 8); ctx.fillStyle = '#ffffff'; ctx.fillRect(20, -1, 10, 2); }
        ctx.restore();

        // ★ 로봇 HP 바 그리기
        ctx.fillStyle = '#c0392b'; ctx.fillRect(t.x - 15, t.y + 20, 30, 4);
        ctx.fillStyle = '#2ecc71'; ctx.fillRect(t.x - 15, t.y + 20, 30 * (t.hp / t.maxHp), 4);
        
        ctx.fillStyle = '#f1c40f'; ctx.font = '12px Arial bold'; ctx.fillText(`Lv.${totalLv}`, t.x - 14, t.y - 20 - (totalLv*2));
        if (gameState.selectedTower === t) { ctx.beginPath(); ctx.arc(t.x, t.y, actualRange, 0, Math.PI * 2); ctx.strokeStyle = `rgba(255, 255, 255, 0.15)`; ctx.stroke(); }
    }

    function triggerGameOver(isQuit = false) {
        gameState.status = 'gameover'; 
        
        for(let i=0; i<60; i++) {
            particles.push({ x: 10 + Math.random()*70, y: Math.random()*650, r: 5 + Math.random()*25, color: '#2c3e50', life: 2.0, decay: 0.005, vx: (Math.random()-0.2)*15, vy: (Math.random()-0.5)*15, growth: 0, isPoly: true });
            if(i%2==0) createFire(10 + Math.random()*60, Math.random()*650, 1.5);
        }
        let msg = isQuit ? "GAME ABORTED" : "SYSTEM FAILURE";
        createFloatingText(canvas.width/2, canvas.height/2, msg, "#e74c3c", 60);
        
        let finalScore = gameState.quizTotal > 0 ? Math.round((gameState.quizCorrect / gameState.quizTotal) * 100) : 0;

        setTimeout(() => { 
            document.getElementById('go-wave').innerText = gameState.wave;
            document.getElementById('go-quiz').innerText = `${gameState.quizCorrect}/${gameState.quizTotal}`;
            document.getElementById('go-score').innerText = `${finalScore}점`;
            
            document.getElementById('go-score').style.color = finalScore >= 80 ? '#2ecc71' : (finalScore >= 50 ? '#f1c40f' : '#e74c3c');
            
            document.getElementById('gameover-overlay').style.display = 'flex';
        }, 2000);
    }

    function createSmoke(x, y, intensity) {
        let c = 80 - Math.floor(intensity * 50); 
        particles.push({ x: x, y: y, r: 5 + Math.random()*10, color: `rgba(${c},${c},${c},0.7)`, life: 1.0, decay: 0.01 + Math.random()*0.01, vx: (Math.random()-0.5)*0.5, vy: -1 - Math.random()*1.5*intensity, growth: 0.2, isText: false });
    }
    function createFire(x, y, intensity) {
        let colors = ['#f1c40f', '#e67e22', '#e74c3c'];
        particles.push({ x: x, y: y, r: 8 + Math.random()*10*intensity, color: colors[Math.floor(Math.random()*colors.length)], life: 1.0, decay: 0.03 + Math.random()*0.03, vx: (Math.random()-0.5)*1.5, vy: -2 - Math.random()*2, growth: -0.2, isText: false });
    }

    function updatePlayingLogic() {
        if (gameState.heatLv > 0 && gameState.baseHp > 0 && gameState.baseHp < gameState.baseMaxHp) {
            gameState.regenProgress += (gameState.heatLv * 0.2); 
            if(gameState.regenProgress >= 100) {
                gameState.regenProgress = 0;
                gameState.baseHp = Math.min(gameState.baseMaxHp, gameState.baseHp + 1);
                createFloatingText(40, canvas.height/2, "+1", "#2ecc71", 20);
            }
        } else { gameState.regenProgress = 0; }

        if (gameState.waveTimerFrames > 0) {
            
            // ★ 웨이브 중간(15초) 퀴즈 발동 로직
            if (gameState.waveTimerFrames <= 15 * 60 && !gameState.midQuizTriggered) {
                gameState.midQuizTriggered = true;
                triggerQuiz();
                return; // 퀴즈 창이 뜨면 이번 프레임 로직 정지
            }

            gameState.waveTimerFrames--;
            if (gameState.frameCount % 60 === 0) updateUI();
            
            let spawnRate = Math.max(12, 80 - (gameState.wave * 3)); 
            if (gameState.frameCount % spawnRate === 0) { spawnNormalEnemy(false); }
            
            // ★ 용착불량 분산 스폰 로직
            if (gameState.fusionToSpawn > 0) {
                let fusionInterval = Math.max(10, Math.floor((30 * 60) / (gameState.wave < 10 ? 10 : gameState.wave)));
                if (gameState.waveTimerFrames % fusionInterval === 0) {
                    spawnNormalEnemy(true); // true = 용착불량 스폰
                    gameState.fusionToSpawn--;
                }
            }

        } else if (!gameState.bossSpawned) {
            updateUI(); 
            spawnBossSwarm();
            gameState.bossSpawned = true;
        } else if (enemies.length === 0) {
            startNextWave(); 
        }

        towers.forEach(t => {
            let spec = TOWER_SPECS[t.type];
            let actualRange = spec.range + (t.upg.voltage * 25);
            let target = enemies.find(e => Math.hypot(e.x - t.x, e.y - t.y) < actualRange);
            
            if (target) {
                t.angle += (Math.atan2(target.y - t.torch_end.y, target.x - t.torch_end.x) - t.angle) * 0.2; 
                if (t.cooldown <= 0) {
                    target.hp -= (spec.baseDmg + (t.upg.current * 15));
                    let sSlow = t.upg.filler * 30;
                    if(sSlow > 0) target.slowTimer = Math.max(target.slowTimer, sSlow);
                    
                    let sRad = spec.splashBase + (t.upg.gas * 20);
                    if (sRad > 0) {
                        let sDmg = t.upg.gas * 10;
                        enemies.forEach(e => { if (e !== target && Math.hypot(e.x - target.x, e.y - target.y) < sRad) { e.hp -= (t.type === 'FCAW' ? (spec.baseDmg*0.5) : 0) + sDmg; if(sSlow > 0) e.slowTimer = Math.max(e.slowTimer, sSlow); } });
                    }
                    projectiles.push({ x: t.torch_end.x, y: t.torch_end.y, target: target, life: 3, color: spec.color });
                    t.cooldown = spec.cd;
                }
            } else { t.angle += (0 - t.angle) * 0.05; }
            t.cooldown--;
        });

        for (let i = enemies.length - 1; i >= 0; i--) {
            let e = enemies[i]; 
            
            if (e.bossRef && enemies.includes(e.bossRef)) {
                e.x = e.bossRef.x + e.offsetX;
                e.y = e.bossRef.y + e.offsetY;
                let angle = Math.atan2(e.offsetY, e.offsetX);
                let dist = Math.hypot(e.offsetX, e.offsetY);
                angle += 0.06; 
                e.offsetX = Math.cos(angle) * dist;
                e.offsetY = Math.sin(angle) * dist;
            } else {
                if (e.bossRef) e.bossRef = null; 
                let currentSpeed = e.speed;
                if(e.slowTimer > 0) { e.slowTimer--; currentSpeed *= 0.4; }
                e.x -= currentSpeed;
            }

            // ★ 용착불량 몬스터 로봇 충돌 자폭 로직
            if (e.type === 'Fusion') {
                let hitTower = towers.find(t => Math.hypot(t.x - e.x, t.y - e.y) < 30); // 충돌 반경
                if (hitTower) {
                    hitTower.hp -= e.dmg;
                    createFire(e.x, e.y, 1.0);
                    enemies.splice(i, 1);
                    updateUI();
                    
                    if (hitTower.hp <= 0) {
                        createEffect(hitTower.x, hitTower.y, '#e74c3c', 50, 1.5, 0.02); // 파괴 이펙트
                        towers = towers.filter(t => t !== hitTower); // 로봇 철거됨 (환불 X)
                        createFloatingText(hitTower.x, hitTower.y - 20, "ROBOT DESTROYED!", "#e74c3c", 16);
                        closeMenus();
                    } else {
                        createFloatingText(hitTower.x, hitTower.y - 20, `HP -${e.dmg}`, "#e74c3c", 14);
                    }
                    continue; // 자폭했으므로 아래 기지 충돌 생략
                }
            }
            
            let hitTargetX = gameState.beadShield > 0 ? 95 : 80;
            if (e.x < hitTargetX) {
                if (gameState.beadShield > 0) { gameState.beadShield -= e.dmg; if(gameState.beadShield < 0) gameState.beadShield = 0; } 
                else { gameState.baseHp -= Math.max(1, e.dmg - (gameState.notchLv * 2)); }
                
                enemies.splice(i, 1); updateUI();
                if(gameState.baseHp <= 0) { triggerGameOver(); return; }
                continue;
            }
        }

        for (let i = enemies.length - 1; i >= 0; i--) {
            let e = enemies[i];
            if(e.hp <= 0) {
                if(e.type === 'Crack') { createEffect(e.x, e.y, '#c0392b', 80, 1.0, 0.02); }
                gameState.resources += e.reward; updateUI(); enemies.splice(i, 1);
            }
        }
    }

    function updateParticleLogic() {
        for (let i = particles.length - 1; i >= 0; i--) {
            let pt = particles[i];
            pt.life -= pt.decay; 
            if(pt.life <= 0) particles.splice(i, 1);
            else {
                pt.r += (pt.growth !== undefined) ? pt.growth : (pt.decay * 1.5);
                if (pt.r < 0.1) pt.r = 0.1;
                pt.x += pt.vx || 0; pt.y += pt.vy || 0;
                if (pt.isPoly) pt.vy += 0.5; 
            }
        }
    }

    function gameLoop() {
        if (gameState.status === 'paused') return;
        
        for(let step = 0; step < gameState.speed; step++) {
            if(gameState.status === 'gameover') break; 
            
            gameState.frameCount++;
            if (gameState.status === 'playing') updatePlayingLogic();
            updateParticleLogic();
        }

        ctx.clearRect(0, 0, canvas.width, canvas.height);

        ctx.fillStyle = '#2c3e50'; ctx.fillRect(0, 0, 80, canvas.height);
        
        let hpRatio = gameState.baseHp / gameState.baseMaxHp;
        let damageLevel = 1.0 - hpRatio; 
        if (gameState.status === 'gameover') damageLevel = 1.5;

        ctx.strokeStyle = '#111'; ctx.lineWidth = 1.5;
        baseCracks.forEach(c => {
            if (hpRatio < c.thresh || gameState.status === 'gameover') {
                ctx.beginPath(); ctx.moveTo(c.pts[0].x, c.pts[0].y);
                for(let i=1; i<c.pts.length; i++) ctx.lineTo(c.pts[i].x, c.pts[i].y);
                ctx.stroke();
            }
        });

        if (damageLevel > 0.3 && gameState.frameCount % Math.max(1, Math.floor(10 - damageLevel*8)) === 0) createSmoke(10 + Math.random()*60, 50 + Math.random()*500, damageLevel);
        if (damageLevel > 0.6 && gameState.frameCount % Math.max(1, Math.floor(6 - damageLevel*5)) === 0) createFire(20 + Math.random()*50, 50 + Math.random()*500, damageLevel);

        ctx.fillStyle = '#ecf0f1'; ctx.font = '24px Arial bold';
        ctx.save(); ctx.translate(45, canvas.height/2 + 80); ctx.rotate(-Math.PI/2); ctx.fillText('WELDING ZONE', 0, 0); ctx.restore();

        if (gameState.beadShield > 0 && gameState.status === 'playing') {
            ctx.fillStyle = '#e67e22'; ctx.fillRect(80, 0, 15, canvas.height);
            ctx.fillStyle = '#d35400';
            for(let i=0; i<canvas.height; i+=20) { ctx.beginPath(); ctx.arc(87, i, 7, 0, Math.PI*2); ctx.fill(); }
        }

        if(gameState.status === 'playing') {
            towers.forEach(t => {
                let spec = TOWER_SPECS[t.type];
                let actualRange = spec.range + (t.upg.voltage * 25);
                drawAndCalculateRobot(ctx, t, spec, actualRange);
            });

            enemies.forEach(e => {
                drawEnemy(ctx, e);
                ctx.fillStyle = '#c0392b'; ctx.fillRect(e.x - 15, e.y - e.radius - 20, 30, 4);
                ctx.fillStyle = '#2ecc71'; ctx.fillRect(e.x - 15, e.y - e.radius - 20, 30 * (e.hp / e.maxHp), 4);
            });
        }

        for (let i = projectiles.length - 1; i >= 0; i--) {
            let p = projectiles[i];
            if (enemies.includes(p.target)) {
                ctx.beginPath(); ctx.moveTo(p.x, p.y);
                let segments = 5; let dx = (p.target.x - p.x)/segments; let dy = (p.target.y - p.y)/segments;
                for(let j=1; j<=segments; j++) { ctx.lineTo(p.x + dx*j + (j===segments?0:(Math.random()-0.5)*15), p.y + dy*j + (j===segments?0:(Math.random()-0.5)*15)); }
                ctx.strokeStyle = '#ffffff'; ctx.lineWidth = 2 + Math.random()*2; ctx.shadowBlur = 10; ctx.shadowColor = p.color; ctx.stroke(); ctx.shadowBlur = 0;
            }
            p.life--; if (p.life <= 0) projectiles.splice(i, 1);
        }

        particles.forEach(pt => {
            ctx.globalAlpha = Math.max(0, pt.life);
            ctx.fillStyle = pt.color;
            if (pt.isText) { ctx.font = `${pt.size}px bold Arial`; ctx.fillText(pt.text, pt.x, pt.y); } 
            else if (pt.isPoly) { ctx.beginPath(); ctx.moveTo(pt.x, pt.y); ctx.lineTo(pt.x+pt.r, pt.y+pt.r/2); ctx.lineTo(pt.x-pt.r/2, pt.y+pt.r); ctx.fill(); } 
            else { ctx.beginPath(); ctx.arc(pt.x, pt.y, pt.r, 0, Math.PI * 2); ctx.fill(); }
        });
        ctx.globalAlpha = 1.0; 

        requestAnimationFrame(gameLoop);
    }
    
    function createEffect(x, y, color, maxR, life = 1.0, decay = 0.05, vx=0, vy=0) { particles.push({ x: x, y: y, r: 2, maxR: maxR, color: color, life: life, decay: decay, vx:vx, vy:vy, isText: false }); }
    function createFloatingText(x, y, text, color, size = 14) { particles.push({ x: x, y: y, text: text, color: color, life: 1.0, size: size, decay: 0.02, isText: true }); }
