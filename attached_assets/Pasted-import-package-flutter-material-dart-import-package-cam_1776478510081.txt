import 'package:flutter/material.dart';
import 'package:camera/camera.dart';

class WeldCameraScreen extends StatefulWidget {
  final List<CameraDescription> cameras;

  const WeldCameraScreen({Key? key, required this.cameras}) : super(key: key);

  @override
  State<WeldCameraScreen> createState() => _WeldCameraScreenState();
}

class _WeldCameraScreenState extends State<WeldCameraScreen> {
  late CameraController _controller;
  bool _isCameraInitialized = false;

  @override
  void initState() {
    super.initState();
    _initCamera();
  }

  // 1. 카메라 초기화 세팅
  Future<void> _initCamera() async {
    // 후면 카메라 선택 (보통 리스트의 0번째)
    _controller = CameraController(
      widget.cameras[0],
      ResolutionPreset.max, // AI 분석을 위해 최고 화질 사용
      enableAudio: false,
    );

    try {
      await _controller.initialize();
      setState(() {
        _isCameraInitialized = true;
      });
    } catch (e) {
      print("카메라 초기화 에러: $e");
    }
  }

  // 2. 사진 촬영 로직
  Future<void> _takePicture() async {
    if (!_controller.value.isInitialized) return;
    if (_controller.value.isTakingPicture) return;

    try {
      XFile picture = await _controller.takePicture();
      // TODO: 촬영된 picture.path를 Replit 백엔드로 전송하는 API 호출 로직 연결
      print("사진 촬영 완료! 경로: ${picture.path}");
      
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('사진이 촬영되었습니다. AI 분석을 시작합니다.')),
      );
    } catch (e) {
      print("사진 촬영 실패: $e");
    }
  }

  @override
  void dispose() {
    _controller.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    if (!_isCameraInitialized) {
      return const Scaffold(
        backgroundColor: Colors.black,
        body: Center(child: CircularProgressIndicator(color: Colors.blue)),
      );
    }

    // 디바이스 화면 크기
    final size = MediaQuery.of(context).size;

    return Scaffold(
      backgroundColor: Colors.black,
      body: Stack(
        children: [
          // [레이어 1] 카메라 프리뷰 (전체 화면)
          SizedBox(
            width: size.width,
            height: size.height,
            child: CameraPreview(_controller),
          ),

          // [레이어 2] UI 가이드라인 오버레이 (반투명 마스크 + 타겟 박스)
          Positioned.fill(
            child: CustomPaint(
              painter: GuideOverlayPainter(),
            ),
          ),

          // [레이어 3] 사용자 안내 텍스트
          Positioned(
            top: 100,
            left: 0,
            right: 0,
            child: Column(
              children: [
                Container(
                  padding: const EdgeInsets.symmetric(horizontal: 16, vertical: 8),
                  decoration: BoxDecoration(
                    color: Colors.black.withOpacity(0.7),
                    borderRadius: BorderRadius.circular(8),
                  ),
                  child: const Text(
                    "ArUco 마커를 우측 상단 네모 괄호에 맞추고\n스마트폰을 모재와 수평으로 유지하세요.",
                    textAlign: TextAlign.center,
                    style: TextStyle(color: Colors.white, fontSize: 16, fontWeight: FontWeight.bold),
                  ),
                ),
              ],
            ),
          ),

          // [레이어 4] 하단 촬영 버튼
          Positioned(
            bottom: 50,
            left: 0,
            right: 0,
            child: Center(
              child: GestureDetector(
                onTap: _takePicture,
                child: Container(
                  height: 80,
                  width: 80,
                  decoration: BoxDecoration(
                    shape: BoxShape.circle,
                    border: Border.all(color: Colors.white, width: 4),
                    color: Colors.white.withOpacity(0.3),
                  ),
                  child: const Center(
                    child: Icon(Icons.camera_alt, color: Colors.white, size: 40),
                  ),
                ),
              ),
            ),
          ),
        ],
      ),
    );
  }
}

// 카메라 위에 타겟 박스와 반투명 배경을 그리는 CustomPainter
class GuideOverlayPainter extends CustomPainter {
  @override
  void paint(Canvas canvas, Size size) {
    final backgroundPaint = Paint()..color = Colors.black.withOpacity(0.4);
    
    // 중앙 용접 비드 가이드 영역 (세로로 긴 직사각형)
    final centerRect = Rect.fromCenter(
      center: Offset(size.width / 2, size.height / 2),
      width: size.width * 0.7,
      height: size.height * 0.5,
    );

    // 우측 상단 ArUco 마커 가이드 영역 (정사각형)
    final markerRect = Rect.fromLTWH(
      centerRect.right - 80, 
      centerRect.top + 20, 
      60, 
      60
    );

    // 전체 화면을 어둡게 칠하고, 가이드 영역만 투명하게 뚫어냄
    canvas.drawPath(
      Path.combine(
        PathOperation.difference,
        Path()..addRect(Rect.fromLTWH(0, 0, size.width, size.height)),
        Path()..addRect(centerRect)..addRect(markerRect),
      ),
      backgroundPaint,
    );

    // 중앙 비드 박스 테두리 (초록색)
    final borderPaint = Paint()
      ..color = Colors.greenAccent
      ..style = PaintingStyle.stroke
      ..strokeWidth = 2.0;
    canvas.drawRect(centerRect, borderPaint);

    // 마커 박스 테두리 (노란색)
    final markerBorderPaint = Paint()
      ..color = Colors.amber
      ..style = PaintingStyle.stroke
      ..strokeWidth = 3.0;
    canvas.drawRect(markerRect, markerBorderPaint);
  }

  @override
  bool shouldRepaint(covariant CustomPainter oldDelegate) => false;
}
