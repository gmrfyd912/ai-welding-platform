【버그 원인 분석】

원인 1 (가장 큰 원인): AI 프롬프트에서 동일 이미지를 high/low 두 장 전송하고
  "두 번째 이미지 기준으로 좌표 보고"라고 지시하지만, GPT-4o는 같은 이미지 두 장을
  구분하지 못하여 좌표 기준점이 혼란됨. 특히 detail:"low"로 보내면 GPT-4o 내부에서
  512x512 영역에 맞추면서 비정사각형 이미지에 패딩이 추가되어 좌표가 틀어짐.

원인 2: 좌표 그리드가 5×5로 너무 거칠어서 (36개 교차점만 가능) 정밀한 위치 지정 불가.

원인 3: 측면/이면 사진이 추가되면 이미지 순서가 바뀌는데, AI가 "두 번째 이미지"를
  잘못 참조하여 엉뚱한 이미지 기준으로 좌표를 보고.

원인 4 (클라이언트): heatmapContainerW가 0일 때 SCREEN_W - 32를 폴백으로 사용하지만,
  실제 컨테이너 폭은 SectionCard 패딩(16*2) + 콘텐츠 패딩(16*2)을 빼야 하므로
  초기 렌더링 시 마커 위치가 어긋남.

원인 5 (클라이언트): 마커 위치가 이미지 영역 밖으로 벗어나는 것을 방지하는
  boundary clamping 로직이 없음.

type ImageContent = {
  type: "image_url";
  image_url: { url: string; detail: "high" | "low" };
};
type TextContent = { type: "text"; text: string };
type MessageContent = ImageContent | TextContent;

function photoContent(base64: string, detail: "high" | "low"): ImageContent {
  return { type: "image_url", image_url: { url: `data:image/jpeg;base64,${base64}`, detail } };
}

// ★ 수정: buildUserMessage - 정면 이미지 1장만 전송 (low 복제본 제거)
function buildUserMessage(params: {
  process: string;
  posture: string;
  material: string;
  selfScore: number;
  plateThickness?: string;
  photos: { front: string; side?: string; back?: string };
  previousResultsSummary?: string;
  adminFeedback?: string;
}): MessageContent[] {
  const content: MessageContent[] = [];

  // ★ 핵심 수정: 정면 사진 1장만 high detail로 전송 (기존: high + low 2장 전송하여 혼란 유발)
  content.push(photoContent(params.photos.front, "high"));

  if (params.photos.side) content.push(photoContent(params.photos.side, "high"));
  if (params.photos.back) content.push(photoContent(params.photos.back, "high"));

  // ★ 수정: 이미지 순서 설명 - "두 번째=전체뷰(좌표기준)" 제거
  let imageCount = 1;
  let photoOrder = "1번=정면";
  if (params.photos.side) { imageCount++; photoOrder += `, ${imageCount}번=측면`; }
  if (params.photos.back) { imageCount++; photoOrder += `, ${imageCount}번=이면`; }

  let text = `【용접 정보】공정:${params.process} / 자세:${params.posture} / 소재:${params.material} / 자체점수:${params.selfScore}점`;
  if (params.plateThickness) text += ` / 부재두께:${params.plateThickness}mm`;
  text += `\n【이미지】총 ${imageCount}장 (${photoOrder})`;
  // ★ 수정: "두 번째 이미지 기준" 대신 "정면 사진(1번 이미지) 기준"으로 명시
  text += `\n※ defectLocations 좌표는 반드시 정면 사진(1번 이미지) 기준으로, 이미지 왼쪽끝=x:0, 오른쪽끝=x:100, 위쪽끝=y:0, 아래쪽끝=y:100 백분율로 보고하십시오.`;
  if (params.adminFeedback) text += `\n\n【관리자 추가 검사기준 - 반드시 적용】\n${params.adminFeedback}`;
  if (params.previousResultsSummary) text += `\n\n【이전 이력 (comprehensiveReport에 반영)】\n${params.previousResultsSummary}`;

  content.push({ type: "text", text });
  return content;
}

function applyBeadWidthThreshold(value: string): "양호" | "주의" | "불량" {
  const maxMatch = value.match(/최대\s*([\d.]+)/);
  const minMatch = value.match(/최소\s*([\d.]+)/);
  if (maxMatch && minMatch) {
    const diff = parseFloat(maxMatch[1]) - parseFloat(minMatch[1]);
    if (diff < 1) return "양호"; if (diff < 2) return "주의"; return "불량";
  }
  const m = value.match(/([\d.]+)\s*mm/);
  if (m) { const v = parseFloat(m[1]); if (v < 1) return "양호"; if (v < 2) return "주의"; return "불량"; }
  return "주의";
}

function applyStraightnessThreshold(value: string): "양호" | "주의" | "불량" {
  const m = value.match(/[±+]?\s*([\d.]+)/);
  if (m) { const d = parseFloat(m[1]); if (d < 1) return "양호"; if (d < 2) return "주의"; return "불량"; }
  return "주의";
}

// ★ 수정: parseDefectLocations - 정수 반올림 추가
function parseDefectLocations(raw: any[]): { name: string; x: number; y: number }[] {
  return raw
    .map((loc: any) => {
      const xNum = parseFloat(String(loc.x));
      const yNum = parseFloat(String(loc.y));
      return {
        name: String(loc.name ?? ""),
        // ★ 수정: Math.round 추가하여 정수 좌표로 정규화
        x: Math.round(Math.min(100, Math.max(0, isNaN(xNum) ? 50 : xNum))),
        y: Math.round(Math.min(100, Math.max(0, isNaN(yNum) ? 50 : yNum))),
      };
    })
    .filter((loc) => loc.name.length > 0);
}

function fixDefects(defectsRaw: any[]): any[] {
  const LIMITS: Record<string, string> = {
    "균열": "불허", "기공": "불허", "언더컷": "0.5mm",
    "오버랩": "불허", "스패터": "경미허용", "아크": "불허",
  };
  return defectsRaw.map((d: any) => {
    const key = Object.keys(LIMITS).find((k) => d.name?.includes(k));
    return { ...d, standard: "선급", limit: key ? LIMITS[key] : d.limit };
  });
}

interface PerPhotoAnalysis {
  beadAnalysis: {
    totalScore: number;
    width: { value: string; score: number; result: string };
    height: { value: string; score: number; result: string };
    straightness: { value: string; score: number; result: string };
    pitchUniformity: { value: string; score: number; result: string };
  };
  defects: any[];
  defectLocations: { name: string; x: number; y: number }[];
}

async function analyzePerPhoto(params: {
  photoBase64: string;
  photoLabel: string;
  process: string;
  posture: string;
  plateThickness?: string;
}): Promise<PerPhotoAnalysis> {
  const response = await openai.chat.completions.create({
    model: "gpt-4o",
    max_tokens: 2000,
    messages: [
      { role: "system", content: PER_PHOTO_SYSTEM_PROMPT },
      { role: "user", content: buildPerPhotoMessage(params) as any },
    ],
  });

  const content = response.choices[0]?.message?.content ?? "";
  const jsonMatch = content.match(/\{[\s\S]*\}/);
  if (!jsonMatch) throw new Error("JSON 추출 실패");

  const parsed = JSON.parse(jsonMatch[0]);
  const widthRaw = parsed.beadAnalysis?.width ?? { value: "측정불가(추정)", score: 50, result: "주의" };
  const straightnessRaw = parsed.beadAnalysis?.straightness ?? { value: "±1.0mm(추정)", score: 50, result: "주의" };
  widthRaw.result = applyBeadWidthThreshold(widthRaw.value);
  straightnessRaw.result = applyStraightnessThreshold(straightnessRaw.value);

  const defects = fixDefects(parsed.defects ?? FALLBACK_DEFECTS);
  const defectLocations = Array.isArray(parsed.defectLocations)
    ? parseDefectLocations(parsed.defectLocations)
    : [];
  console.log(`[${params.photoLabel}] 좌표:`, JSON.stringify(defectLocations));

  return {
    beadAnalysis: {
      totalScore: Math.min(100, Math.max(0, Math.round(parsed.beadAnalysis?.totalScore ?? 50))),
      width: widthRaw,
      height: parsed.beadAnalysis?.height ?? { value: "측정불가(추정)", score: 50, result: "주의" },
      straightness: straightnessRaw,
      pitchUniformity: parsed.beadAnalysis?.pitchUniformity ?? { value: "측정불가(추정)", score: 50, result: "주의" },
    },
    defects,
    defectLocations,
  };
}

export function registerWeldAnalysisRoute(app: Express): void {
  app.post("/api/analyze-weld", largeBodyParser, async (req: Request, res: Response) => {
    const { photos, imageBase64, process, posture, material, selfScore, previousResultsSummary, plateThickness } = req.body;

    const frontPhoto = photos?.front || imageBase64;
    if (!frontPhoto) return res.status(400).json({ error: "정면 사진(front)이 필요합니다." });

    try {
      let adminFeedback: string | undefined;
      try {
        const fbRows = await pool.query("SELECT feedback_text FROM admin_feedback ORDER BY created_at DESC LIMIT 30");
        if (fbRows.rows.length > 0) {
          adminFeedback = fbRows.rows.map((r: any, i: number) => `${i + 1}. ${r.feedback_text}`).join("\n");
        }
      } catch {}

      const procStr = process || "FCAW";
      const postStr = posture || "1G";
      const matStr = material || "탄소강 평판";
      const thick = plateThickness || undefined;

      const [mainResponse, sideAnalysis, backAnalysis] = await Promise.all([
        openai.chat.completions.create({
          model: "gpt-4o",
          max_tokens: 4000,
          messages: [
            { role: "system", content: SYSTEM_PROMPT },
            {
              role: "user",
              content: buildUserMessage({
                process: procStr, posture: postStr, material: matStr,
                selfScore: selfScore ?? 50, plateThickness: thick,
                photos: { front: frontPhoto, side: photos?.side, back: photos?.back },
                previousResultsSummary, adminFeedback,
              }) as any,
            },
          ],
        }),
        photos?.side
          ? analyzePerPhoto({ photoBase64: photos.side, photoLabel: "측면", process: procStr, posture: postStr, plateThickness: thick })
            .catch((e) => { console.error("측면 분석 오류:", e); return null; })
          : null,
        photos?.back
          ? analyzePerPhoto({ photoBase64: photos.back, photoLabel: "이면", process: procStr, posture: postStr })
            .catch((e) => { console.error("이면 분석 오류:", e); return null; })
          : null,
      ]);

      const content = mainResponse.choices[0]?.message?.content ?? "";
      console.log("정면 AI 응답 (앞 500자):", content.slice(0, 500));

      const jsonMatch = content.match(/\{[\s\S]*\}/);
      if (!jsonMatch) {
        console.error("GPT 응답 JSON 추출 실패:", content.slice(0, 200));
        return res.json(buildFallbackResult(selfScore ?? 50));
      }

      const parsed = JSON.parse(jsonMatch[0]);

      const widthRaw = parsed.beadAnalysis?.width ?? { value: "측정불가(추정)", score: 50, result: "주의" };
      const straightnessRaw = parsed.beadAnalysis?.straightness ?? { value: "±1.0mm(추정)", score: 50, result: "주의" };
      widthRaw.result = applyBeadWidthThreshold(widthRaw.value);
      straightnessRaw.result = applyStraightnessThreshold(straightnessRaw.value);

      const fixedDefects = fixDefects(parsed.defects ?? FALLBACK_DEFECTS);

      const frontLocations = Array.isArray(parsed.defectLocations)
        ? parseDefectLocations(parsed.defectLocations)
        : [];
      console.log("정면 좌표:", JSON.stringify(frontLocations));

      const frontAnalysis: PerPhotoAnalysis = {
        beadAnalysis: {
          totalScore: Math.min(100, Math.max(0, Math.round(parsed.beadAnalysis?.totalScore ?? 50))),
          width: widthRaw,
          height: parsed.beadAnalysis?.height ?? { value: "측정불가(추정)", score: 50, result: "주의" },
          straightness: straightnessRaw,
          pitchUniformity: parsed.beadAnalysis?.pitchUniformity ?? { value: "측정불가(추정)", score: 50, result: "주의" },
        },
        defects: fixedDefects,
        defectLocations: frontLocations,
      };

      const photoAnalyses: Record<string, PerPhotoAnalysis> = { front: frontAnalysis };
      if (sideAnalysis) photoAnalyses.side = sideAnalysis;
      if (backAnalysis) photoAnalyses.back = backAnalysis;

      const result = {
        aiScore: Math.min(100, Math.max(0, Math.round(parsed.aiScore ?? 50))),
        overallVerdict: parsed.overallVerdict === "PASS" ? "PASS" : "FAIL",
        beadAnalysis: frontAnalysis.beadAnalysis,
        defects: fixedDefects,
        defectLocations: frontLocations,
        photoAnalyses,
        improvements: Array.isArray(parsed.improvements) ? parsed.improvements : ["분석 데이터를 확인해주세요."],
        comprehensiveReport: parsed.comprehensiveReport ?? "",
        top3Defects: Array.isArray(parsed.top3Defects) ? parsed.top3Defects : [],
      };

      res.json(result);
    } catch (err) {
      console.error("용접 AI 분석 오류:", err);
      res.json(buildFallbackResult(selfScore ?? 50));
    }
  });
}
