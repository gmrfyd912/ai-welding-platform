import numpy as np

def calculate_weld_score(bead_width_max, bead_width_min, defects):
    # 비드 폭 편차 계산 및 점수화
    width_variance = bead_width_max - bead_width_min
    if width_variance < 1.0:
        shape_score, shape_eval = 95, "우수"
    elif 1.0 <= width_variance < 2.0:
        shape_score, shape_eval = 85, "양호"
    elif 2.0 <= width_variance < 3.0:
        shape_score, shape_eval = 70, "주의"
    else:
        shape_score, shape_eval = 50, "불량"
        
    # 7대 결함 감점 로직 (선급 기준 반영)
    penalty = 0
    defect_list_ko = []
    is_critical_fail = False
    
    for defect in defects:
        d_type = defect.get('class', '').lower()
        if d_type == 'crack':
            penalty += 100
            defect_list_ko.append("균열(-100점, 즉시 불합격)")
            is_critical_fail = True
        elif d_type in ['lack_of_fusion', 'lack_of_penetration', '용착불량']:
            penalty += 20
            defect_list_ko.append("용입/용착 불량(-20점)")
        elif d_type == 'porosity': penalty += 10; defect_list_ko.append("기공(-10점)")
        elif d_type == 'undercut': penalty += 10; defect_list_ko.append("언더컷(-10점)")
        elif d_type == 'overlap': penalty += 10; defect_list_ko.append("오버랩(-10점)")
        elif d_type == 'arc_strike': penalty += 10; defect_list_ko.append("아크 스트라이크(-10점)")
        elif d_type == 'spatter': penalty += 5; defect_list_ko.append("스패터(-5점)")

    final_score = max(0, min(100, shape_score - penalty))
    is_pass = "FAIL" if (is_critical_fail or final_score < 70) else "PASS"
    
    return {
        "final_score": final_score, "is_pass": is_pass,
        "width_variance": round(width_variance, 2), "shape_eval": shape_eval,
        "detected_defects": defect_list_ko
    }

from openai import AsyncOpenAI
import json

client = AsyncOpenAI(api_key="YOUR_OPENAI_API_KEY")

async def get_expert_advice(image_base64: str, weld_data: dict):
    system_prompt = "당신은 30년 경력의 CWI 용접 검사관입니다. 제공된 수치를 바탕으로 원인 분석과 교정 방법만 JSON으로 응답하세요."
    user_content = f"최종점수: {weld_data['final_score']}, 편차: {weld_data['width_variance']}mm, 결함: {weld_data['detected_defects']}. 이 데이터를 바탕으로 사진을 분석해줘."

    response = await client.chat.completions.create(
        model="gpt-4o",
        response_format={ "type": "json_object" },
        messages=[
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": [
                {"type": "text", "text": user_content},
                {"type": "image_url", "image_url": {"url": f"data:image/jpeg;base64,{image_base64}", "detail": "high"}}
            ]}
        ]
    )
    return json.loads(response.choices[0].message.content)
