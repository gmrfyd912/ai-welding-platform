1. 파일 생성
파일명: test_client.py

역할: 로컬(또는 Replit 내부) FastAPI 서버에 실제 용접 사진을 전송하고, 반환된 분석 결과(점수, 수치, GPT 리포트)를 터미널에 예쁘게 출력함.

2. 코드 내용
아래 파이썬 코드를 한 글자도 빠짐없이 test_client.py에 작성해.

Python
import requests
import json
import os

# 1. FastAPI 서버 주소 (Replit 내부에서 실행할 것이므로 localhost:8000 사용)
API_URL = "http://0.0.0.0:8000/analyze-welding"

# 2. 테스트할 이미지 파일명 (이 파일이 Replit 폴더에 업로드되어 있어야 함)
IMAGE_FILE = "sample_weld.jpg"

def run_test():
    # 이미지 파일 존재 여부 확인
    if not os.path.exists(IMAGE_FILE):
        print(f"❌ 에러: '{IMAGE_FILE}' 파일이 보이지 않습니다.")
        print("Replit 왼쪽 파일 목록에 테스트용 사진을 업로드하고 이름을 'sample_weld.jpg'로 변경해 주세요.")
        return

    print(f"🚀 분석 시작: {IMAGE_FILE}를 서버로 전송 중...")
    
    try:
        # 파일 열기 및 전송
        with open(IMAGE_FILE, 'rb') as f:
            files = {'file': (IMAGE_FILE, f, 'image/jpeg')}
            response = requests.post(API_URL, files=files)
        
        # 결과 출력
        if response.status_code == 200:
            result = response.json()
            print("\n" + "="*50)
            print("✅ AI 용접 분석 결과")
            print("="*50)
            print(f"📊 최종 점수: {result.get('final_score', 'N/A')}점 ({result.get('is_pass', 'N/A')})")
            print(f"📏 비드 폭 편차: {result.get('width_variance', 'N/A')}mm")
            print(f"🔍 감지된 결함: {', '.join(result.get('detected_defects', [])) if result.get('detected_defects') else '없음'}")
            print("-" * 50)
            print("🧠 명장의 조언 (GPT-4o):")
            # GPT 리포트가 dictionary 형태일 경우와 string 형태일 경우 모두 대응
            advice = result.get('expert_report', {})
            if isinstance(advice, dict):
                print(f"📍 원인분석: {advice.get('cause_analysis', '내용 없음')}")
                print(f"🔧 교정방법: {advice.get('action_plan', '내용 없음')}")
            else:
                print(advice)
            print("="*50)
        else:
            print(f"❌ 서버 에러 ({response.status_code}): {response.text}")

    except Exception as e:
        print(f"❌ 통신 중 오류 발생: {e}")

if __name__ == "__main__":
    run_test()
